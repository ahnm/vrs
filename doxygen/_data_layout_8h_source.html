425  if (mappedDataLayout_ != nullptr) {
426  return mappedDataLayout_->getFixedData<T>(offset, size);
427  }
428  if (offset != kNotFound && offset + size <= fixedData_.size()) {
429  return reinterpret_cast<T*>(fixedData_.data() + offset);

446  template <class T>
447  T* getVarData(size_t varPieceIndex, size_t& outCount) {
448  if (mappedDataLayout_ != nullptr) {
449  return mappedDataLayout_->getVarData<T>(varPieceIndex, outCount);
450  }
451  if (varPieceIndex < varSizePieces_.size()) {
452  const IndexEntry& indexEntry = getVarSizeIndex()[varPieceIndex];
453  if (indexEntry.getOffset() + indexEntry.getLength() <= varData_.size()) {
454  outCount = indexEntry.getLength() / sizeof(T);
455  return reinterpret_cast<T*>(varData_.data() + indexEntry.getOffset());
456  }
457  }
458  outCount = 0;
459  return nullptr;
460  }
